 方式一：
 const query = Bmob.Query('tableName');
 query.set('id', 'objectId') //需要修改的objectId
 query.set('nickName', 'Bmob后端云')
 query.save().then(res => {
 console.log(res)
 }).catch(err => {
 console.log(err)
 })

const query = Bmob.Query('tableName');

}).catch(err => {
